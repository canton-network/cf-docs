final case class PackageMetadata(interfaces: Set[Identifier] = Set.empty, templates: Set[Identifier] = Set.empty, interfacesImplementedBy: InterfacesImplementedBy = Map.empty, packageNameMap: Map[PackageName, PackageResolution] = Map.empty, packageIdVersionMap: Map[PackageId, (PackageName, PackageVersion)] = Map.empty, packages: Map[PackageId, PackageSignature] = Map.empty) extends Product with Serializable

**Summary**
